generate_global_forbes_list = {
	scope = player
	
	is_shown = { 
		always = yes
	}

	is_valid = {
		always = yes
	}

	ai_is_valid = {
		always = no
	}

	effect = {
		set_variable = {
			name = forbes_mode
			value = 1
		}
		generate_forbes_list = yes
	}
}

open_forbes_list = {
	scope = player
	
	is_shown = { 
		always = yes
	}

	is_valid = {
		always = yes
	}

	ai_is_valid = {
		always = no
	}

	effect = {
		if = {
			limit = { 
				root = { 
					NOT = { has_variable = forbes_mode } 
				} 
			} 
			set_variable = {
				name = forbes_mode
				value = 2
			}
		}
		generate_forbes_list = yes
	}
}
