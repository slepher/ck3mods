	value = 0

			limit = {
				vassal_contract_obligation_level:religious_rights != 0
			}

		if = {
			limit = {
				vassal_contract_obligation_level:fortification_rights != 0
			}
			add = 1
		}
		if = {
			limit = {
				vassal_contract_obligation_level:coinage_rights != 0
			}

		}
		if = {
			limit = {

		if = {

			}
			add = 1
		}
	}
}

		add = 1
	}

overheld_title = {
	value = 0
	if = {
		limit = {
			held_county > 1
		}
		add = held_county
		subtract = 1
	}
	if = {
		limit = {
			held_duchy > 1
		}
		add = held_duchy
		subtract = 1
	}
	if = {
		limit = {
			held_kingdom > 1
		}
		add = held_kingdom
		subtract = 1
	}
}

military_strength = {
	value = 0
	add = max_military_strength
}

claim_values = {
	value = 0
	every_claim = {
		limit = {
			NOT = {
				holder = {
					OR = {
						target_is_same_character_or_above = root
						target_is_vassal_or_below = root
					}
				}
			}
		}
		if = {
			limit = {
				tier = tier_empire
			}
			add = 1000
		}
		else_if = {
			limit = {
				tier = tier_kingdom
			}
			add = 100
		}
		else_if = {
			limit = {
				tier = tier_duchy
			}
			add = 10
		}
		else_if = {
			limit = {
				tier = tier_county
			}
			add = 1
		}
	}
}
