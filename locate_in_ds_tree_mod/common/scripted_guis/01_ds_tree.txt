ds_tree_locate = {
	scope = dynasty

	is_shown = { 
		always = yes
	}

	is_valid = {
		always = yes
	}

	ai_is_valid = {
		always = no
	}

	effect = {
		root.var:ds_tree_selected = {
			save_scope_as = current_character
		}
		debug_log = "ds tree locate"
		up_character_location = yes
	}
}

ds_tree_init = {
	scope = dynasty

	saved_scopes = {
		current_character
	}

	is_shown = { 
		always = yes
	}

	is_valid = {
		always = yes
	}

	ai_is_valid = {
		always = no
	}

	effect = {
		root = {
			set_variable = {
				name = ds_tree_selected
				value = scope:current_character
			}
		}
		debug_log = "ds tree init"
		up_character_location = yes
	}
}
